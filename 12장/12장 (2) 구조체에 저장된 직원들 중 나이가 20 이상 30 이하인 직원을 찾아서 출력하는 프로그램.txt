#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

// 직원 구조체 정의
typedef struct {
    int number;
    char name[21];  // 이름 최대 20글자 + null 문자
    char phone[31]; // 전화번호 최대 30글자 + null 문자
    int age;
} employee;

int main() {
    employee employees[10]; // 최대 10명의 직원
    int n;

    // 직원 수 입력
    scanf("%d", &n);

    // 직원 정보 입력
    for (int i = 0; i < n; i++) {
        printf("번호 이름 전화번호 나이: ");
        scanf("%d %s %s %d", &employees[i].number, employees[i].name, employees[i].phone, &employees[i].age);
    }

    // 나이가 20 이상 30 이하인 직원 출력
    for (int i = 0; i < n; i++) {
        if (employees[i].age >= 20 && employees[i].age <= 30) {
            printf("%s\n", employees[i].name);
        }
    }

    return 0;
}
