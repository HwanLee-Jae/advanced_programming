제일 큰 요소 값의 메모리 주소를 반환하는 max함수를 구현하라.
(같은 수가 입력되지 않는다고 가정하자.)
Main 함수는 수정 불가하다.

#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <limits.h>

int* min(int array[], int* address) {
	int* min_ptr = &array[0];
	int length = address - array + 1; // 배열 길이 계산

	for (int i = 0; i < length; i++) {
		if (array[i] < *min_ptr) {
			min_ptr = &array[i];
		}
	}

	return min_ptr; // 최솟값의 주소 반환
}

int* max(int* address, int array[]) {
	int* max_ptr = address;
	int length = array - address + 1; // 배열 길이 계산

	for (int i = 0; i < length; i++) {
		if (address[i] > *max_ptr) {
			max_ptr = &address[i];
		}
	}

	return max_ptr; // 최댓값의 주소 반환
}
